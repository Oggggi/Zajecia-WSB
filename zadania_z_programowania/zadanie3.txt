internal class Program
{
    private static void Main(string[] args)
    {
        int liczbaElementow;
        int[] tablica;
        Console.WriteLine("Podaj liczbe elementow do wprowadzenia: ");
        liczbaElementow = int.Parse(Console.ReadLine());
        while (liczbaElementow <= 0)
        {
            Console.WriteLine("Liczba elementow musi byc wieksza od zera. Podaj ponownie: ");
            liczbaElementow = int.Parse(Console.ReadLine());
        }
        tablica = new int[liczbaElementow];
        for (int i = 0; i < liczbaElementow; i++)
        {
            Console.WriteLine($"Podaj element {i + 1}: ");
            tablica[i] = int.Parse(Console.ReadLine());
        }
        Console.Write("Elementy wprowadzonej tablicy: ");
        for (int i = 0; i < liczbaElementow; i++)
        {
            Console.Write(tablica[i] + " ");
        }
    }
}

{
    private static void Main(string[] args)
    {
        Random random = new Random();
        byte[] tablica = new byte[10];
        random.NextBytes(tablica);
        int lp = 1;
        int minimum = tablica[0];
        for (int i = 1; i < tablica.Length; i++)
        {
            if (tablica[i] < minimum)
            {
                minimum = tablica[i];
                lp = i;
            }
        }
        for (int i = 0; i < tablica.Length; i++)
        {
            Console.WriteLine("Tablica: " + "[" + i + "] " + tablica[i]);
        }
        Console.WriteLine("Najmniejsza wartosc w tablicy to: " + minimum + " o indeksie: " + lp);
    }
}

internal class Program
{
    private static void Main(string[] args)
    {
        int[,] tablica = new int[5, 5];
        Random random = new Random();
        int suma = 0;
        for (int i = 0; i < tablica.GetLength(0); i++)
        {
            for (int j = 0; j < tablica.GetLength(1); j++)
            {
                tablica[i, j] = random.Next(0, 9);
            }
        }
        for (int i = 0; i < tablica.GetLength(0); i++)
        {
            for (int j = 0; j < tablica.GetLength(1); j++)
            {
                Console.Write(tablica[i, j] + " ");
            }
            Console.WriteLine();
        }
        for (int i = 0; i < tablica.GetLength(0); i++)
        {
            suma += tablica[i, i];
            suma += tablica[i, tablica.GetLength(1) - 1 - i];
        }
        Console.WriteLine("Suma przekatnych: " + suma);
    }
}

//zadanie 3.6
internal class Program
{
    private static void Main(string[] args)
    {
        char[] tablica = new char[8];
        int binarna;
        int dziesietna;
        Console.WriteLine("Podaj liczbe w systemie binarnym (maksymalnie 8 znakow): ");
        tablica = Console.ReadLine().ToCharArray();
        while (tablica.Length > 8)
        {
            Console.WriteLine("Liczba jest za duza. Podaj liczbe w systemie binarnym (maksymalnie 8 znakow): ");
            tablica = Console.ReadLine().ToCharArray();
        }
        foreach (char znak in tablica)
        {
            if (znak != '0' && znak != '1')
            {
                Console.WriteLine("Podano niepoprawny znak");
                return;
            }
        }
        Array.Reverse(tablica);
        dziesietna = 0;
        for (int i = 0; i < tablica.Length; i++)
        {
            binarna = (int)Char.GetNumericValue(tablica[i]);
            dziesietna += binarna * (int)Math.Pow(2, i);
        }
        Console.WriteLine("Liczba w systemie dziesietnym: " + dziesietna);
    }
}

        }
        tekstUzytkownika = tekstUzytkownika.Trim();
        for (int i = 0; i < tekstUzytkownika.Length; i++)
        {
            if (!(tekstUzytkownika[i] == ' ' && tekstUzytkownika[i + 1] == ' '))
            {
                tekstZredagowany += tekstUzytkownika[i];
            }
        }
        Console.WriteLine("Zredagowany tekst: ");
        Console.WriteLine(tekstZredagowany);
    }
}

            iloscSlow = tekstZredagowany.Split(' ').Length;

        }
        else
        {
            Console.WriteLine("Nie podano żadnego tekstu.");
        }
    }
}

//zadanie 3.11
internal class Program
{
    private static void Main(string[] args)
    {
        string tekstPierwszy, tekstDrugi, tekstTrzeci = "";

        Console.WriteLine("Podaj pierwszy tekst (zapamietaj dlugosc tekstu): ");
        tekstPierwszy = Console.ReadLine();
        Console.WriteLine("Podaj drugi tekst (dlugosc tekstu taka sama jak pierwszego): ");
        tekstDrugi = Console.ReadLine();

        while (tekstDrugi.Length != tekstPierwszy.Length)
        {
            Console.WriteLine("Dlugosc drugiego tekstu jest inna niz pierwszego. Podaj ponownie: ");
            tekstDrugi = Console.ReadLine();
        }
        for (int i = 0; i < tekstPierwszy.Length; i++)
        {
            if (tekstPierwszy[i] == tekstDrugi[i])
            {
                tekstTrzeci += tekstPierwszy[i];
            }
            else
            {
                tekstTrzeci += "?";
            }
        }
        if (tekstTrzeci == tekstPierwszy)
        {
            Console.WriteLine("Teksty sa identyczne.");
        }
        else
        {
            Console.WriteLine("Roznica: " + tekstTrzeci);   
        }
    }
}
